<script>
function filterEvents(club) {
    const events = document.querySelectorAll('.event');
    events.forEach(event => {
        if (event.getAttribute('data-club') === club || club === 'all') {
            event.style.display = 'flex';
        } else {
            event.style.display = 'none';
        }
    });
}
</script>
